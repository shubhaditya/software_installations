Linux:

%%% Unpack .tar file
tar -xvf openmpi-1.10.0.tar 

#%%% Build a dummy folder in main directory to run ./configure etc. from inside it
cd path/openmpi-1.10.0
mkdir openmpi-build
cd openmpi-build 

#%%% Run config (with --prefix flag for a local build and CC, FC to specify compilers, e.g. CC=/home-research/shubhadm/packages/gcc_local/bin/gcc. Probably also need to specify CCFLAGS/FCFLAGS if the 
#%%% compiler gives an error, usually due to library linking e.g. FCFLAGS=-Wl,-rpath,/home-research/shubhadm/packages/gcc_local/lib64)

../configure --prefix=/full_path_to_local_directory CC=/full_path_to_compiler FC=/full_path_to_compiler CCFLAGS=flag_cc FCFCLAGS=flag_fc

#%%% Make and install
make all install
